        function sendToWhatsApp() {
            // ضع رقم الواتساب الخاص بك هنا بكود الدولة بدون أصفار أو علامة +
            // مثال: للرقم المصرى 01000000000 اكتب 201000000000
            const myPhoneNumber = "201000000000"; 

            // جلب القيم من النموذج
            const name = document.getElementById('fullName').value;
            const phone = document.getElementById('whatsapp').value;
            const service = document.getElementById('serviceType').value;
            const dateTime = document.getElementById('dateTime').value;
            const pickup = document.getElementById('pickup').value;
            const dropoff = document.getElementById('dropoff').value;
            const passengers = document.getElementById('passengers').value;
            const flight = document.getElementById('flightNumber').value || "لا يوجد";

            // التأكد من ملء الحقول الإلزامية قبل الإرسال
            if(!name || !phone || !service || !dateTime || !pickup || !dropoff || !passengers) {
                alert("برجاء ملء كافة الحقول الأساسية قبل تأكيد الحجز.");
                return;
            }

            // تنسيق نص الرسالة
            const message = `*طلب حجز جديد من موقع Taxi Hub* \n\n` +
                            `👤 *الاسم:* ${name}\n` +
                            `📞 *الواتساب:* ${phone}\n` +
                            `🗺️ *نوع الخدمة:* ${service}\n` +
                            `📅 *التاريخ والوقت:* ${dateTime}\n` +
                            `📍 *مكان الاستلام:* ${pickup}\n` +
                            `🏁 *الوجهة:* ${dropoff}\n` +
                            `👥 *عدد الأفراد:* ${passengers}\n` +
                            `✈️ *رقم الرحلة الجوية:* ${flight}`;

            // تحويل النص إلى رابط صالح للمتصفح
            const whatsappUrl = `https://wa.me/${201021023002}?text=${encodeURIComponent(message)}`;

            // فتح الواتساب في نافذة جديدة
            window.open(whatsappUrl, '_blank');
        }
    </script>
